# run install script
./install.sh

# install formatters : xmllint and jq
dnf install  xmllint  jq

sd          # search and replace
snippets-ls # snippets lsp (https://github.com/mgholam/snippets-ls)
tmux
lazygit + delta # side by side diff
uncrustify  # c# formatting
biome       # js formatting
